                        'macro', 'miscellanea', 'nature','holidays',

                        'men','minimalism', 'rendering', 'situations', 'dog',  
                        'sports', 'style','textures', 'fantasy', 'films',  
                        'flowers', 'anime', 'avto'
  - -t TIME, --time TIME  image time: 'day', 'week', '17day', 'month'
  - -n NAME, --name NAME  set filename
